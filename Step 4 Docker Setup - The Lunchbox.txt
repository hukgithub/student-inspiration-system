Step 4: Docker Setup - The Lunchbox
Create docker-compose.yml to run everything together:

yaml:

version: '3.8'
services:
  db:
    image: postgres:13
    environment:
      POSTGRES_PASSWORD: super_secret_password  # Change this!
  backend:
    build: ./backend
    command: python manage.py runserver 0.0.0.0:8000
    volumes:
      - ./backend:/app
    ports:
      - "8000:8000"
    depends_on:
      - db
  frontend:
    build: ./frontend
    command: npm start
    ports:
      - "3000:3000"
    depends_on:
      - backend
  redis:
    image: redis:alpine
